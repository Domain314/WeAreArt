CREATE DATABASE waa_contact;
CREATE TABLE `waa_contact`.`contact` ( `name` VARCHAR(50) NULL DEFAULT NULL , `email` VARCHAR(50) NULL DEFAULT NULL , `comment1` VARCHAR(5000) NULL DEFAULT NULL , `comment2` VARCHAR(5000) NULL DEFAULT NULL , `checkbox` BOOLEAN NULL DEFAULT NULL , `timestamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , PRIMARY KEY (`timestamp`)) ENGINE = InnoDB;
